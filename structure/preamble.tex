% Inserts a picture as a figure with a reference 
\newcommand{\pix}[3]{
	\begin{figure}[!htb]
	\begin{center}
	\includegraphics[width=#2cm]{pictures/#1}
	\end{center}
	\caption{\label{fig:#1} #3}
	\end{figure}
}

\newcommand{\onePic}[4]{
	\begin{figure}[h!]
		\begin{center}
			\includegraphics[width=#4\linewidth]{#1}
		\end{center}

	\end{figure}
}

}

\newcommand{\dbpix}[4]{
	\begin{figure}[!htb]
	\begin{center}
	\begin{minipage}{\linewidth}  % Choose here the width of the minipage
	  \begin{minipage}{0.5\linewidth}
		\includegraphics[width=\linewidth]{#1.jpg}
	  \end{minipage}%
	  \hspace{\fill}%
	  \begin{minipage}{0.5\linewidth}%
		\includegraphics[width=\linewidth]{#2.jpg}  
	  \end{minipage}
	\end{minipage}
	\end{center}
	\caption{\label{#3} #4}
	\end{figure}
}
